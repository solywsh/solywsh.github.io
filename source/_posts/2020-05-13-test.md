# 对10个整数排序

```c
#include<stdio.h>
int main()
{
	int i,j,num[10],temp;
	printf("请输入10个数：\n");
	for ( i = 0; i < 10; i++)
	{
		scanf("%d", &num[i]);
	}
	for ( i = 0; i < 9; i++)
	{
		for ( j = 0; j < 9 - i; j++)
		{
			if (num[j] > num[j + 1])
			{
				temp = num[j];
				num[j] = num[j + 1];
				num[j + 1] = temp;
			}
		}
	}

	for ( i = 1; i < 10; i++)

		printf(" %d", num[i]);
	}

}
```

# 最大公约数和最小公倍数

 写两个函数，分别求两个整数的最大公约数和最小公倍数，用主函数调用这两个函数，并输出结果，两个整数由键盘输入：

```c
#include <stdio.h>
int hcf(int x, int y)
{
	int temp, rem;
	if (x < y)
	{
		temp = x;
		x = y;
		y = temp;
	}
	rem = x % y;
	while (rem != 0)
	{
		x = y;
		y = rem;
		rem = x % y;
	}
	return (y);
}

int lcd(int x, int y)
{
	int Lcd;
	Lcd = (x * y) / hcf(x, y);
	return (Lcd);
}

int main()
{
	int num_1, num_2;
	scanf("%d %d", &num_1, &num_2);
	printf("%d %d", hcf(num_1, num_2), lcd(num_1, num_2));
	return 0;
}
```

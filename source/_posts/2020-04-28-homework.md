# 年月日

**题目内容：**定义一个结构体变量（包括年、月、日）。计算该日在本年中是第几天，注意闰年问题

**输入格式:   **年 月 日

**输出格式：**第几天

**输入样例：**2020 2 29

**输出样例：**60

{% note info %}
闰年是公历中的名词。闰年分为普通闰年和世纪闰年。
普通闰年:公历年份是4的倍数的，且不是100的倍数，为普通闰年。（如2004年就是闰年）；

<span style="background-color:#FFF000">世纪闰年:公历年份是整百数的，必须是400的倍数才是世纪闰年（如1900年不是世纪闰年，2000年是世纪闰年）；</span>

世纪闰年:公历年份是整百数的，必须是400的倍数才是世纪闰年（如1900年不是世纪闰年，2000年是世纪闰年）；
闰年(Leap Year)是为了弥补因人为历法规定造成的年度天数与地球实际公转周期的时间差而设立的。补上时间差的年份为闰年。闰年共有366天（1-12月分别为31天，29天，31天，30天，31天，30天，31天，31天，30天，31天，30天，31天）。
{% endnote %}

```c
#include<stdio.h>
typedef struct {
	int year;
	int month;
	int day;
}Date;
int main()
{
	Date input;
	scanf("%d %d %d", &input.year, &input.month, &input.day);
	void days(Date count);
	days(input);
	return 0;
}
void days(Date count)
{
	int a[12] = { 31,28,31,30,31,30,31,31,30,31,30,31 };
	int sum = count.day, i = 0;//先给sum赋值，等于这个月的第几天
	for (i = 0; i < count.month - 1; i++)//再计算在这个月之前有几天
		sum += a[i];
	if (count.month==2)
		/*如果本月为2月的话
		由于已经给sum赋值这月的天数count.day了
		不用再次赋值,直接打印*/
	{
		printf("%d", sum);
	}
	else
	{
		if (count.year % 4 == 0 && count.year % 100 != 0)//普通闰年
			printf("%d", sum + 1);
		else if (count.year % 400 == 0)//世纪闰年
			printf("%d", sum + 1);
		else
			printf("%d", sum);
	}
	return;
}
```

# 平均成绩

**题目内容：**有3个学生，每个学生的数据包括学号（整数）、3门课程的成绩（浮点数），从键盘输入10个学生数据，要求输出每个学生3门课程的平均成绩（保留两位小数），同时输出平均成绩最高的学生学号

**输入格式:**

学生一学号 成绩一 成绩二 成绩三

学生二学号 成绩一 成绩二 成绩三

学生三学号 成绩一 成绩二 成绩三

**输出格式：**

`学生一平均成绩` `学生二平均成绩` `学生三平均成绩`（无空格）

`成绩最高的学生学号`

**输入样例：**

101 80 80 80

102 90 90 90

103 100 90 90

**输出样例：**

**80.00 90.00 93.33**

**103**

``` c
#include <stdio.h>
#define n 3
typedef struct
{
    char num[10];
    int score[3];
    float ave;
}STU;
int main()
{
    STU stu[10], max;
    int i, j;
    int sum[3] = { 0,0,0 };
    for (i = 0; i < n; i++)
    {
        scanf("%s %d %d %d", stu[i].num, &stu[i].score[0], &stu[i].score[1], &stu[i].score[2]);
        sum[0] += stu[i].score[0];
        sum[1] += stu[i].score[1];
        sum[2] += stu[i].score[2];
        stu[i].ave = (stu[i].score[0] + stu[i].score[1] + stu[i].score[2]) / 3.0;
    }
    printf("%.2f %.2f %.2f\n", stu[0].ave, stu[1].ave, stu[2].ave);
    max = stu[0];
    for (j = 1; j < n; j++)
    {
        if (stu[j].ave > max.ave)
            max = stu[j];
    }
    printf("%s", max.num);
    return 0;
}
```

```c
#include<stdio.h>
#include<stdlib.h>
#define LEN sizeof(LinkList)
typedef struct node {
    int num;
    struct node* next;
}LinkList;
int n;
```

```c
LinkList* creat(void)
{
```

```c
    int i = 0;
    p1 = p2 = (LinkList*)malloc(LEN);
    scanf("%d", &n);
    for (i = 0; i < n; i++)
    {
        scanf("%d", &p1->num);
        if (i == 0)head_1 = p1;
        else p2->next = p1;
        p2 = p1;
        p1 = (LinkList*)malloc(LEN);
    }
    p2->next = NULL;
    return (head_1);
}


void print(LinkList* head_2)
{
    if (head_2 != NULL)
    {
        printf("%d", head_2->num);
        head_2 = head_2->next;
        for (; head_2 != NULL; head_2 = head_2->next)
        {
            printf(" %d", head_2->num);
        }
    }
}

LinkList *delLink(LinkList* head_3) {            //链表的删除操作
    /* 这里使用了指针，本来返回值是void的，但是常常出现内存上的一些错误
    所以建议还是吧返回值改回来
    这样理解时相对简单一些
    这里全部返回头部 head
    */
    LinkList *r, * q;
    int Num,i;
    scanf("%d", &Num);
    for (q = head_3, i = 1; i < Num;  q = q->next, i++);//注意这里直接分号结束了

    if (q == head_3) //如果删除第一个节点 
    {
        r= q->next;
        free(q);
        return (r);//以r作为头部返回
    }

    else 
    {                                    //删除其他节点 
        for (r = head_3; r->next != q; r = r->next);
        if (r->next != NULL) {
            r->next = q->next;
            free(q);
            return (head_3);//由于不是第一个节点，所以作为头部的第一个节点并没有更改，直接返回
        }
    }
}

int main(void)
{
    LinkList* head, temp;
    head = creat();
    head = delLink(head);
    print(head);
    return 0;
}
```

# 向链表中插入元素

**题目内容：**在前面两题的基础上，新建一个函数实现向链表中插入元素的功能

**输入格式:**

新建链表的长度n

要插入元素的<span style="background:#00c758">值</span> 要插入元素的<span style="background:#00c758">位置</span>

顺序输出插入后的链表节点的值（注意最后一个数字后无空格）

**输入样例：**

4

1 2 3 4

5 3

**输出样例：**

**1 2 3 5 4**

```c
LinkList * insertList(LinkList* head_3) {     //链表的添加操作，首位置为 0 处
    LinkList *r, * q;
    int place,i;
    LinkList *ins;
    ins = (LinkList *)malloc(LEN);
    scanf("%d %d",&ins->num,&place);

    for (q = head_3, i = 1; i < place;  q = q->next, i++);

注意上边for循环取巧了

    {
        ins->next = head_3;
        return (ins);
    }
    else 
    {                                //添加在其他位置
        r = q->next;
        q->next = ins;
        ins->next = r;
        return (head_3);
    }
    
}

int main(void)
{
    LinkList* head, temp;
    head = creat();
    head = insertList(head);
    print(head);
    return 0;
}
```

# 归并链表

**题目内容：**

输入两个升序链表，加他们归并为一个有序链表

链表1长度

链表1

归并链表

```c
LinkList* Combine(LinkList* head1, LinkList* head2)
{
    LinkList* head = NULL, * p, * tail;
    tail = (LinkList*)malloc(LEN);
    while (head1 && head2)
    {
        if (head1->num < head2->num)
        {
            p = head1;
            head1 = head1->next;
        }
        else
        {
            p = head2;
            head2 = head2->next;
        }
        p->next = NULL;
        if (!head)
            head = tail = p;
        else
        {
            tail = tail->next = p;
        }
    }
    if (head1)
        p = head1;
    else
        p = head2;
    tail->next = p;
    return head;
```
